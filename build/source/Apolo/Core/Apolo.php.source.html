  56  /**
  57   * Apolo framework
  58   * 
  59   * Just a litle framework that runs over routes mapped to controllers.
  60   *
  61   * This framework works in three basic steps:
  62   * - create routes
  63   * - create controller
  64   * - create a view
  65   *
  66   * The routes defines where all URLs in the project. Each URL need to be
  67   * redirected to an Controller. It is defined using
  68   * {@see Apolo\Apolo::setRoutes()}.
  69   *
  70   * The controller is a simple class. This class do not needs to extends any
  71   * other to work. Any method accessed represents the method in the controller.
  72   *
  73   * To be easy, the method is mapped with the same name inside the controller.
  74   * So if the user access a page via GET, Apolo will use the get method in the
  75   * controller and if the user sends a request inside a form that uses POST,
  76   * Apolo will use the post method of the controller.
  77   *
  78   * Here is a simple sample of a simple controller:
  79   *
  80   * <code>
  81   * class Hello
  82   * {
  83   *     public function get()
  84   *     {
  85   *         return  '&lt;form method="post">'
  86   *               . ' &lt;input type="text" name="name" />'
  87   *               . ' &lt;input type="submit" />'
  88   *               . '&lt;/form>';
  89   *     }
  90   *     public function post()
  91   *     {
  92   *         $name = filter_input(
  93   *             INPUT_POST, 'login', FILTER_SANITIZE_FULL_SPECIAL_CHARS
  94   *         );
  95   *         return 'Hello ' . $name;
  96   *     }
  97   * }
  98   * </code>
  99   *
 100   * @category  Core
 101   * @package   Apolo
 102   * @author    Michael Castillo <michaelgranados@gmail.com>
 103   * @copyright 2012 Michael Castillo
 104   * @license   http://www.opensource.org/licenses/mit-license.php  MIT License
 105   * @version   GIT: <git_id>
 106   * @link      http://github.com/ApoloPHP/Core
 107   */

 154      /**
 155       * Set new routes
 156       *
 157       * Set routes for redirect the user to any controller. You need to pass
 158       * the param of routes as array like this:
 159       *
 160       * <code>
 161       * $routes = array(
 162       *   'route' => 'Controller',
 163       * );
 164       * </code>
 165       *
 166       * All routes will be added to the actual routes and no one route setted
 167       * before will be replaced.
 168       *
 169       * You can pass a route like a regexp to especify a set of routes, like
 170       * this:
 171       *
 172       * <code>
 173       * $routes = array(
 174       *   'publish/\d+/?' => 'Publish',
 175       * );
 176       * </code>
 177       *
 178       * Here, you can use the replacer strings to use in your regexp, so the
 179       * development will be more intuictive.
 180       *
 181       * <code>
 182       * $routes = array(
 183       *   'see/:digit:/:slug:/?' => 'See',
 184       * );
 185       * </code>
 186       *
 187       * In the aboe sample, <kbd>:digit:</kbd> will be replaced by a
 188       * <kbd>[0-9]+</kbd> regexp. Here is the list of replacers:
 189       *
 190       * <pre>
 191       * - /              => \/
 192       * - :alpha:        => [a-zA-Z]+
 193       * - :alphanumeric: => [a-zA-Z0-9]+
 194       * - :digit:        => [0-9]+
 195       * - :slug:         => [a-zA-Z0-9_-]+
 196       * </pre>
 197       *
 198       * All elements in parentesis will be an argument to your controller like
 199       * this:
 200       *
 201       * <code>
 202       * $routes = array(
 203       *     '/edit/(article|image)/(:digit:)' => 'Edit',
 204       * //               arg1     ,  arg2
 205       * );
 206       * </code>
 207       *
 208       * So your controller will be something like this:
 209       *
 210       * <code>
 211       * class Edit
 212       * {
 213       *     function get($area, $id)
 214       *     {
 215       *         // your development
 216       *     }
 217       * }
 218       * </code>
 219       *
 220       * You can set 
 221       *
 222       * @param string[]  $routes Array of routes
 223       * @param string    $type   Type of insertion of routes
 224       *
 225       * @uses   \Apolo\Core\Route
 226       * @access public
 227       * @static
 228       * @return void
 229       */
 230      public static function setRoutes(
 231          array $routes, $type = Route::MODE_REPLACE
 232      ) {
 233          Route::map($routes, $type);
 234      }
